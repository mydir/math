54 template <typename F>
55 inline double de_integrator(const F& f, double a, double b, double tre,
56  double tae) {
57  using std::fabs;
58  using std::log;
59 
60  // Apply the linear change of variables x = ct + d
61  // $$\int_a^b f(x) dx = c \int_{-1}^1 f( ct + d ) dt$$
62  // c = (b-a)/2, d = (a+b)/2
63  double c = 0.5 * (b - a);
64  double d = 0.5 * (a + b);
65  int num_function_evaluations;
66 
67  tae /= c;
68 
69  // Offsets to where each level's integration constants start.
70  // The last element is not a beginning but an end.
71  static int offsets[] = {1, 4, 7, 13, 25, 49, 97, 193};
72  int num_levels = sizeof(offsets) / sizeof(int) - 1;
73 
74  double new_contribution = 0.0;
75  double integral = 0.0;
76  double previous_integral = 0.0;
77  double error_estimate = DBL_MAX;
78  double h = 1.0;
79  double previous_delta, current_delta = DBL_MAX;
80 
81  integral = f(c * de_abcissas[0] + d) * de_weights[0];
82 
83  int i;
84  for (i = offsets[0]; i != offsets[1]; ++i)
85  integral += de_weights[i]
86  * (f(c * de_abcissas[i] + d) + f(-c * de_abcissas[i] + d));
87 
88  for (int level = 1; level != num_levels; ++level) {
89  h *= 0.5;
90  new_contribution = 0.0;
91  for (i = offsets[level]; i != offsets[level + 1]; ++i)
92  new_contribution
93  += de_weights[i]
94  * (f(c * de_abcissas[i] + d) + f(-c * de_abcissas[i] + d));
95  new_contribution *= h;
96 
97  // difference in consecutive integral estimates
98  previous_delta = current_delta;
99  current_delta = fabs(0.5 * integral - new_contribution);
100  previous_integral = integral;
101  integral = 0.5 * integral + new_contribution;
102 
103  // Once convergence kicks in, error is approximately squared
104  // at each step.
105  // Determine whether we're in the convergent region by looking
106  // at the trend in the error.
107  if (level == 1)
108  // previous_delta meaningless, so cannot check
109  // convergence.
110  continue;
111 
112  // Exact comparison with zero is harmless here. Could possibly
113  // be replaced with a small positive upper limit on the size
114  // of current_delta, but determining that upper limit would be
115  // difficult. At worse, the loop is executed more times than
116  // necessary. But no infinite loop can result since there is
117  // an upper bound on the loop variable.
118  if (current_delta == 0.0)
119  break;
120  // previous_delta != 0 or would have been kicked out
121  // previously
122  double r = log(current_delta) / log(previous_delta);
123 
124  if (r > 1.9 && r < 2.1) {
125  // If convergence theory applied perfectly, r would be 2 in
126  // the convergence region. r close to 2 is good enough. We
127  // expect the difference between this integral estimate and
128  // the next one to be roughly delta^2.
129  error_estimate = current_delta * current_delta;
130  } else {
131  // Not in the convergence region. Assume only that error is
132  // decreasing.
133  error_estimate = current_delta;
134  }
135 
136  if (error_estimate < 0.1 * tae
137  && fabs(1 - integral / previous_integral) < tre)
138  break;
139  }
140 
141  num_function_evaluations = 2 * i - 1;
142  error_estimate *= c;
143  return c * integral;
144 }
